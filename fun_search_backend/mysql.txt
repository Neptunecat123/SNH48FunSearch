CREATE TABLE team_info(
        id INT AUTO_INCREMENT PRIMARY KEY,
        name VARCHAR(25),
        group_name VARCHAR(25),
        city VARCHAR(25),
        start_time VARCHAR(50),
        disband_time VARCHAR(50),
        status INT
)

CREATE TABLE `election_info`(
        `id` INT AUTO_INCREMENT PRIMARY KEY,
        `date` VARCHAR(50),
        `member` VARCHAR(50),
        `rank` INT,
        `votes` INT,
        `gname` VARCHAR(10),
        `tname` VARCHAR(10)
)ENGINE=InnoDB DEFAULT CHARSET=utf8


CREATE TABLE `b50_info`(
        `id` INT AUTO_INCREMENT PRIMARY KEY,
        `rank` INT,
        `unit` VARCHAR(200),
        `date` VARCHAR(50),
        `number` INT,
        `members` VARCHAR(500)
)ENGINE=InnoDB DEFAULT CHARSET=utf8
